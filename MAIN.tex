		{m O{\empty} m}
		{\forall\, {#1} \ifthenelse{\equal{#2}{}}{}{\in{#2}} \,.\, {#3}}
	\NewDocumentCommand{\xsome}
		{m O{\empty} m}
		{\exists\, {#1} \ifthenelse{\equal{#2}{}}{}{\in{#2}} \,.\, {#3}}

		{m O{\empty} m}
		{\exists\;\!!\, {#1} \ifthenelse{\equal{#2}{}}{}{\in{#2}} \,.\, {#3}}
	\NewDocumentCommand{\xlam}
		{m O{\empty} m O{\empty}}
		{\lambda\, {#1} \ifthenelse{\equal{#2}{}}{}{\in{#2}} \,.\, {#3} \ifthenelse{\equal{#4}{}}{}{\in{#4}}}

		{O{auto} m O{\empty} m}
